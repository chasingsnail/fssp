地址簿

AWSUI地址簿组件为用户提供了人员地址簿、部门地址簿两种地址簿可供用户选择使用，其中人员地址簿中，可分为普通模式和高级模式，高级模式的回填内容较于普通模式要更丰富。

API
  • $(el).address(config);

config 参数如下：

参数 说明 类型 默认值
callback 回调函数 function -
filter
addressType:地址簿类型，可选值"user"（人员地址簿）、"dept"（部门地址簿）
isAdvMode：是否启用高级模式
addressSetting:地址簿设置，其值为对象(详解如下)
addressSetting
rootDetpId：根部门Id。值类型为String
isDisplayMap：是否显示兼职,默认值为false
layerFrom 值类型为String
layerTo 值类型为String
range：分类显示范围，值为"department(部门)|role(角色)|team(团队)"
choiceType：值为"single"(单选)/"multiple"(多选)
leafType：叶子节点类型。值为"user"(用户)/"dept"(部门)
filterClass：过滤事件Class
sourceField:字典的数据源字段,通常是orguser表中的字段,回填值有
"COMPANYNAME"(单位名称)
"COMPANYID"(单位ID)
"COMPANYNO"(单位编码)
"DEPTNAME"(部门名称)
"DEPTID"(部门ID)
"DEPTNO"(部门编码)
"USERID"(用户ID)
"UID"(用户帐号)
"USERNAME"(用户姓名)
"USERNAMEALIAS"(用户全名)
"USERNO"(员工代码)
"ROLEID"(角色ID),
"POSITIONNAME"(职位名称)
"POSITIONNO"(职位编号)
"POSITIONLAYER"(职位等级)
"EMAIL"(邮箱)
"OFFICETEL"(电话)
"MOBILE"(手机)
"OFFICEFAX"(传真)
"EXT1"(扩展标记1)
"EXT2"(扩展标记2)
"EXT3"(扩展标记3)
"EXT4"(扩展标记4)
"EXT5"(扩展标记5)
"TEAMID"(团队ID(仅在回填团队时生效))
"TEAMNAME"(团队名称(仅在回填团队时生效))
object -
deptTargetField 部门回填字段 string -
isLiveSearch 是否允许搜索 boolean true
allowAnyValue 是否允许输入任意值 boolean false
基本

最基本的用法。

HTML 代码

<div class="awsui-code-demo">
    <input type="text" id="address1" class="awsui-input"/>
</div>

JavaScript 代码

$("#address1").address();
高级地址簿

高级地址簿。

HTML 代码

<div class="awsui-code-demo">
    <input type="text" id="address1_1" class="awsui-input"/>
</div>

JavaScript 代码

var options = {
	filter : {
		addressType: "user",
		isAdvMode: true,
		addressSetting: {
			layerFrom: "1",
			layerTo: "3",
			range: "department|role|team",
			delimiter: " ",
			choiceType: "single",
			leafType: "user"
		},
		sourceField : "UID",
		targetField : "address1_1"
	}
}；
$("#address1_1").address(options);

地址簿完整配置

地址簿完整配置。

HTML 代码

<div class="awsui-code-demo">
    <input type="text" id="address_full" class="awsui-input"/>
</div>

JavaScript 代码

var options = {
	callback : null,
	filter : {
		addressType: "dept",//地址簿类型
		isAdvMode: true,//是否启用高级模式
		addressSetting: {
			rootDetpId: "",
			isDisplayMap: true,
			isDisplayOtherMap: false,
			layerFrom: "",
			layerTo: "",
			range: "department|role|team",//department:部门,role:角色,team:团队
			delimiter: " ",
			choiceType: "single",//single:单选; multiple:多选
			leafType: "user",//叶子节点类型,user:用户;dept:部门
			filterClass: ""//过滤事件
		},
		//字典的数据源字段，通常是orguser表中的字段 ,多个用逗号隔开;回填值选择：
		sourceField: "",
			COMPANYNAME,//单位名称
			COMPANYID,//单位ID,
			COMPANYNO,//单位编码
			DEPTNAME,//部门名称
			DEPTID,//部门ID
			DEPTNO,//部门编码
			USERID,//用户ID
			UID,//用户帐号
			USERNAME,//用户姓名
			USERNAMEALIAS,//用户全名
			USERNO,//员工代码
			ROLEID,//角色ID
			POSITIONNAME,//职位名称
			POSITIONNO,//职位编号
			POSITIONLAYER,//职位等级
			EMAIL,//邮箱
			OFFICETEL,//电话
			MOBILE,//手机
			OFFICEFAX,//传真
			EXT1,//扩展标记1
			EXT2,//扩展标记2
			EXT3,//扩展标记3
			EXT4,//扩展标记4
			EXT5,//扩展标记5
		//回填字段 targetField应与sourceField个数相匹配
		targetField: "",
		deptSourceField : "", //部门数据源字段
			COMPANYNAME,//单位名称
			COMPANYID,//单位ID,
			COMPANYNO,//单位编码
			DEPTNAME,//部门名称
			DEPTID,//部门ID
			DEPTNO,//部门编码
			DEPTZONE,//区域划分
			EXT1,//扩展标记A
			EXT2,//扩展标记B
		deptTargetField : ""  //部门回填字段
	},
	valueType : 0, //0:aliasname; 1:uid; 2:username
	separator : " ", //逗号 空格……分隔符
	inDialog : false,//是否在dialog内部
	dialogId : "",//若地址簿在iframe下,应当配置dialogId来定位回填多个target
	isGrid : false, //在grid中打开
	gridId : "", //初始化grid的jquery对象
	gridRowIndx : 0,//grid行索引
    isClearData: true,//默认提供清空按钮
    isLiveSearch: true, // 快速搜索
    maxRowNumber: 0, // div最大显示几行，大于0生效
    clearCallback: null
};
$("#address_full").address(options);
