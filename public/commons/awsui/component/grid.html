表格方法

方法 说明 用例
destroy 清除AWS Grid的功能 $("selector").awsGrid("destroy")
disable 禁用AWS Grid的功能 $("selector").awsGrid("disable")
enable 启用AWS Grid的功能 $("selector").awsGrid("enable")
editCell 将一个单元格设为编辑状态 $("selector").awsGrid("editCell",{rowIndx:2, colIndx:3})
getCell 获取单元格返回td对象 $("selector").awsGrid("getCell",{rowIndx: 2,colIndx: 3})
getCellIndices 获取已知单元格的索引,返回一个行索引和列索引的对象 var obj = $("selector").awsGrid( "getCellIndices", { $td: $td } );
var colIndx=obj.colIndx,rowIndxPage=obj.rowIndxPage;
getColIndx 获取dataIndex属性的列的索引 $("selector").awsGrid("getColIndx",{dataIndx:dataIndx })
getEditCell 获取正在编辑的单元格,返回null说明没有正在编辑的单元格 var obj = $("selector").awsGrid( "getEditCell" );var $td=obj.$td;var $cell=obj.$cell;
getEditCellData 获取正在编辑的单元格中已/未保存的数据,返回null说明没有正在编辑的单元格 var dataCell = $("selector").awsGrid( "getEditCellData" )
getRow 已知行的索引获取整个行，返回一个TR元素 var $tr = $("selector").awsGrid( "getRow",{rowIndxPage:2})
getRowData 通过索引或id或rowData获取整个行数据 var $tr = $("selector").awsGrid("getRowData",{rowIndx:rowIndx,rowIndxPage:rowIndxPage,recId:recId,rowData:rowData})
addRow 添加行，rowData:可以是数组,rowIndx:插入的位置,track是否标记为修改 var $tr = $("selector").awsGrid( "addRow",{rowData:rowData,rowIndx:2,track:true})
deleteRow 根据行索引删除行（若为false则不存储编辑数据） var $tr=$("selector").awsGrid("deleteRow",{rowIndx:2,track:true}))
commit 提交修改过的行的数据 var $tr=$("selector").awsGrid("commit",{ type:"add",rows:rows})
updateRow 根据行索引更新行 var $tr = $("selector").awsGrid( "updateRow",{rowIndx:rowIndx,row:{ dataIndx1:value1,dataIndx2:value2,...},track:track})
getChanges 获取未保存且编辑过的行的数据 var $tr = $("selector").awsGrid( "getChanges")
isEditableCell 根据全局和列的设置来检查单元格是否可编辑 $("selector").awsGrid( "isEditableCell",{colIndx:3})
quitEditMode 单元格退出编辑状态 $("selector").awsGrid( "quitEditMode" )
option 以键值对的方式获取当前awsGrid的配置信息 var options = $("selector").awsGrid( "option" )
option 设置与提供的属性名称相关的属性值 $("selector").awsGrid( "option", "disabled", true )
option 为AWS Grid设置一个或者多个配置属性 $("selector").awsGrid( "option", {disabled: true} );
refresh 刷新AWS Grid $("selector").awsGrid( "refresh" )
refreshColumn 通过列的索引或键名刷新整个列 $("selector").awsGrid( "refreshColumn", {colIndx:2} )
refreshDataAndView 刷新AWSGrid中的数据视图 $("selector").awsGrid( "refreshDataAndView" );
refreshRow 刷新行,rowIndx和rowIndxPage都起作用 $("selector").awsGrid( "refreshRow", {rowIndx:21} )
saveEditCell 保存当前正在编辑的单元格 $("selector").awsGrid( "saveEditCell")
setSelection 选中/清除行或单元格 $("selector").awsGrid( "setSelection",{rowIndx:rowIndx,rowIndxPage:rowIndxPage,colIndx:colIndx,focus:focus})

表格事件

事件名 说明 用例

beforeSort 对数据进行排序之前调用，如果返回false，排序行为终止 $( ".selector" ).awsGrid({beforeSort: function( event, ui ) {}})

cellKeyDown 单元格按下键盘键的时候触发 $( ".selector" ).awsGrid({cellKeyDown:function(event,ui){}});
cellSave AWS Grid中的单元格被保存后触发 $(".selector").awsGrid({cellSave: function( event, ui ) {}})
cellSelect 选中单元格的时候触发 $( ".selector" ).awsGrid({cellSelect:function(event,ui){}});
cellUnSelect 取消选中单元格的时候触发 $(".selector").awsGrid({cellUnSelect: function( event, ui ) {}});
load 当加载完远程数据的时候执行 $(".selector").awsGrid({load: function( event, ui ){}});
create 在AWS Grid创建后执行,如果是远程数据，方法要放到加载事件中 $(".selector").awsGrid({create: function( event, ui ) {}});
quitEditMode 当一个单元格退出编辑状态时执行 $(".selector").awsGrid({quitEditMode: function( event, ui ) {}});
refresh 网格刷新完之后触发 $( ".selector" ).awsGrid({refresh: function( event, ui ) {}});
rowClick 单击一行后触发。它的事件顺序小于cellclick/td> $( ".selector" ).awsGrid({rowClick: function( event, ui ) {}})

rowUnSelect 取消选中一行后触发此事件 $(".selector").awsGrid({rowUnSelect: function( event, ui ) {}});
sort Grid中用户排序完成后触发 $(".selector").awsGrid({sort: function( event, ui ){}});
