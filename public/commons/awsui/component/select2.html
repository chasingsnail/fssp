列表选择器

优雅的将数据以下拉列表形式展现，弹出一个下拉菜单给用户选择操作，用于代替原生的选择器，或者需要一个更优雅的多选器时

  • 当选项过多时，使用下拉菜单展示并选择内容。
API
  • $(el).select2(config)

config 参数如下：

参数 说明 类型 默认值
width select选择器的宽度 number -
minimumInputLength 最小数量的字符 number -
maximumInputLength 最大数量的字符 number -
minimumResultsForSearch 最小数量的结果,设置-1时搜索功能不出现 number -
maximumSelectionLength 最大选择数目 number -
placeholder 选择框默认文字 string -
separator 分隔符字符或字符串用来划定id string -
allowClear 是否允许清空,若为true必须有placeholder，否则报错 boolean false
multiple 是否允许选择多个值 boolean false
query 函数用于搜索词的查询结果 function -
ajax 选择内置的ajax查询功能。 这个对象作为快捷方式有手动编写一个函数,执行ajax请求 object -
data 数据源,写法为data=[{id:"2",text:"关羽"}] array []
selected 数据源中标明选中项data=[{id:"2",text:"关羽","selected": true}] boolean false
language 语言，汉语为zh-CN string zh-CN
dropdownCss 下拉列表的一个样式,可以设置一个CSS的JSON结构 function/object -
dropdownCssClass css类将被添加到select2下拉的容器 string -
dropdownAutoWidth 当设置为true,尝试自动尺寸下拉基于内容的宽度 boolean -
sorter 排序 function -
templateResult 自定义样式（选项、结果） function -
closeOnSelect 选择后，关闭列表 boolean -
theme 主题，可选项有classic string -
disabled 是否禁用 boolean false

方法说明如下：

方法名 说明
open 打开列表 $eventSelect.on("select2:open",function(e){});
close 关闭列表 $eventSelect.on("select2:close", function (e) {});
select 选择事件 $eventSelect.on("select2:select",function(e){});
unselect 取消选择事件 $eventSelect.on("select2:unselect", function (e) {});
change 切换事件 $eventSelect.on("change", function (e) {});

名称 单选 多选
小三角 有 无
标签化 否 是
搜索框 额外，默认只能按照text搜索 自身，默认只能按照text搜索
默认选中 第一个 无
能否清空 默认没有 可以手动清空

cursor pointer text
基本使用

基本的select2。

HTML 代码

<select id="selectBasic1"></select>
<select id="selectBasic2"></select>

JavaScript 代码

$(document).ready(function() {
    var opt = {
		width : 300,
		data : [
			{id:"lb",text:"刘备"},
			{id:"gy",text:"关羽"},
			{id:"zf",text:"张飞",disabled:true},
			{id:"cc",text:"曹操"}
		]
	};
	$("#selectBasic1").select2($.extend({}, opt, {minimumResultsForSearch:-1}));
	$("#selectBasic2").select2($.extend({}, opt, {disabled:true}));
});

$(document).ready(function() {
	function formatState (state) { // select2的前缀实现方法
		if (!state.id) { return state.text; }
		var $state = $("" + state.img + " " + state.text + "");
		return $state;
	}
	var opt = {
		width : 300,
		templateResult: formatState,
		multiple : true,
		data : [
			{id:"1",text:"Arkansas",img:"<img class='img-flag' src='../img/select2_template_1.png'/>"},
			{id:"2",text:"New Mexico",img:"<img class='img-flag' src='../img/select2_template_2.png'/>"}
		]
	};
	$("#selectTemplateResult").select2(opt);
});

$(document).ready(function() {
    var opt = {
		width : 300,
		data : [
			{id:"lb",text:"刘备"},
			{id:"gy",text:"关羽"},
			{id:"zf",text:"张飞"},
			{id:"cc",text:"曹操"}
		],
		placeholder: "请选择",
        allowClear: true
	};
	$("#selectValue1").select2(opt);
	$("#setValue1").on("click",function(){
		$("#selectValue1").val(["gy"]).trigger("change"); //单选赋值
	});
	$("#getValue1").on("click",function(){
		alert("单选值：" + $("#selectValue1").val());
	});
	var opt1 = {
		multiple : true
	};
	$("#selectValue2").select2($.extend({}, opt, opt1));
	$("#setValue2").on("click",function(){
		$("#selectValue2").val(["gy","zf"]).trigger("change"); //多选赋值
	});
	$("#getValue2").on("click",function(){
		alert("多选值：" + $("#selectValue2").val());
	});
});

禁用效果

禁用效果,添加属性disabled。


HTML 代码

<select id="selectDisabled"></select><br/>
<button id="disabled" type="button" class="awsui-btn awsui-btn-blue">disabled</button>
<button id="enabled" type="button" class="awsui-btn awsui-btn-danger">enabled</button>

JavaScript 代码

$(document).ready(function() {
    var opt = {
		width : 300,
		data : [
			{id:"lb",text:"刘备"},
			{id:"gy",text:"关羽"},
			{id:"zf",text:"张飞"},
			{id:"cc",text:"曹操"}
		]
	};
	$("#selectDisabled").select2(opt);
	$("#disabled").on("click",function(){
		$("#selectDisabled").prop("disabled", true);
		$("#selectDisabled").select2(opt);
		var id = "selectDisabled";
		$("#select2-"+id+"-container").addClass('select2-container--disabled');
	});
	$("#enabled").on("click",function(){
		$("#selectDisabled").prop("disabled",false);
		$("#selectDisabled").select2(opt);
		var id = "selectDisabled";
		$("#select2-"+id+"-container").removeClass('select2-container--disabled');
	});
});
