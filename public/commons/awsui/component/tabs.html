选项卡

提供平级的区域将大块内容进行收纳和展现，保持界面整洁，选项卡片通过JavaScript控制，封装了增加，删除Tab等操作功能；提供了二种选项卡，分别用于不同的场景。

  • 标准线条式页签，用于容器内部的主功能切换，这是最常用的 Tabs。
  • 卡片式的页签，提供可关闭的样式，常用于容器顶部。
API
  • awsui.tabs.init($(el), config);

config 参数如下：

参数 说明 类型 默认值
contentPanel 若tab页下有内容(div、iframe)，此项参数是目标容器 object -
tabPosition 页签位置，选择值有：top、bottom string top
focusShowAfter 点击tab的回调函数:focusShowAfter:function(item){ if(item.index == "tabTwo") alert("显示后触发"); } function -
onClick 点击tab触发的事件 function -
onCloseTab 关闭Tab执行事件的功能 function -
height tab的高度 number -

awsui.tabs.init($(el), config).method

-
方法 说明 用例
addTab 添加一个Tab页签
var setting = {
  item:{
    url:"地址",     title:"标题",     index:index,     content:"主体接受任何布局", },
  contextMenu:false, //是否显示右键菜单,默认值为false
  contextMenuTarget:$(el), //需要的菜单目标
  close:false, //是否显示关闭图标,默认值为true
  onCloseTab:function(){} //关闭Tab执行事件的功能
} 		obj.addTab(setting);
focusTab 定位当前活动的Tab obj.focusTab()
getTabItem 获得Tab项 obj.getTabItem(index)
getTabFirst 获得第一个Tab obj.getTabFirst()<
getTabLast 获得最后一个Tab obj.getTabLast()
existsTab 判断指定的Tab是否存在 obj.existsTab(false)
setTitle 给指定的Tab,设置Titleobj.setTitle(index,title)

$(document).ready(function() {
	var awsuiTabs;
	awsuiTabs = awsui.tabs.init($("#cardTabs"), {
		height:35,
		contentPanel : $("#cardTabsPanel"),
		tabPosition : "top",
		onClick:function(tab){
			return true;
		}
	});
	awsuiTabs.addTab({
		item : {
			color:"#108ee9",
			index : 1,
			title : "<span class='awsui-tab-curent-text'>Tabs1</span>",
			content:$("<div class='awsui-tab-content-item' index='1'>Content of Tab Pane 1</div>")
		},
		close : false
	});
	awsuiTabs.addTab({
		item : {
			index : 2,
			title : "<span class='awsui-tab-curent-text'>Tabs2</span>",
			content:$("<div class='awsui-tab-content-item' index='2'>Content of Tab Pane 2</div>")
		},
		close : false
	});
	awsuiTabs.addTab({
		item : {
			index : 3,
			title : "<span class='awsui-tab-curent-text'>Tabs3</span>",
			content:$("<div class='awsui-tab-content-item' index='3'>Content of Tab Pane 3</div>"),
			contextMenu:true,
			contextMenuTarget:"contextMenu1"
		},
		close : false
	});
});

受控操作示例。

HTML 代码

<button id="addTabs" class="awsui-btn awsui-btn-blue">增加Tab</button>

<button id="toFirseTabs" class="awsui-btn awsui-btn-blue">定位到第一个项</button>
<button id="toLastTabs" class="awsui-btn awsui-btn-blue">定位到最后一项</button>
<div class="awsui-public-border awsui-public-radius awsui-margin-top-10">
	<div class="awsui-tab" id="cardTabsEvent" >
	</div>
	<div id="cardTabsEventPanel" class="aws-simple-tab-content" style="height:50px;">
	</div>
</div>

JavaScript 代码

$(document).ready(function() {
	var awsuiTabsEvent,index=1;
	awsuiTabsEvent = awsui.tabs.init($("#cardTabsEvent"), {
		height:35,
		contentPanel : $("#cardTabsEventPanel"),
		tabPosition : "top",
		onClick:function(tab){
			return true;
		},
		focusShowAfter: function (tab) {
			console.log(tab.index);
		}
	});
	awsuiTabsEvent.addTab({
		item : {
			color:"#108ee9",
			index : 1,
			title : "Tabs1",
			content:$("
Content of Tab Pane 1
")
		},
		close : false
	});
	awsuiTabsEvent.addTab({
		item : {
			index : 2,
			title : "Tabs2",
			content:$("
Content of Tab Pane 2
")
		},
		close : false
	});
	$("#addTabs").on('click',function () {
		addTab();
	});
	$("#removeTabs").on('click',function () {
		removeTab();
	});
	$("#toFirseTabs").on('click',function () {
		toFirst();
	});
	$("#toLastTabs").on('click',function () {
		toLast();
	});
	function addTab(){
		var time = (new Date()).valueOf();
		var setting = {
			item:{
				color:"#3080D9",
				url:"",
				title:""+ index++ +"",
				index:index+1,
				content:$("
"+time+"
")
			},
			close:false,
			onCloseTab:function(){
			},
			focusShowAfter: function (tab) {
				awsuiTabsEvent.setTitle(tab.index,"凯文沃克"+tab.index);
			}
		};
		awsuiTabsEvent.addTab(setting);
	}
	function removeTab(){
		var index = awsuiTabsEvent.tabContainer.find(".current").attr("index");
		awsuiTabsEvent.closeTab(index);
	}
	function toFirst(){
		var obj = awsuiTabsEvent.getTabFirst();
		awsuiTabsEvent.focusTab(obj);
	}
	function toLast(){
		var obj = awsuiTabsEvent.getTabLast();
		awsuiTabsEvent.focusTab(obj);
	}
});
