树形控件

文件夹、组织架构、生物分类、国家地区等等，世间万物的大多数结构都是树形结构。使用树形控件是对企业的组织、部门、人员、流程等其他事物之间的层级关系完美展现，并具有展开收起选择等交互功能。

API
  • awsui.tree.init($(el), config));

config 参数如下：

参数 说明 类型 默认值
showLine 是否显示tree的连线 boolean true
showIcon 是否显示tree节点的图标 boolean true
checkbox 是否启用tree节点的checkbox boolean false
checkInherit 是否启用父子节点选中继承事件,checkbox为true时有效 boolean false
animate 是否启用展开关闭的动画 boolean false
animateTime 启用动画后的动画时间 string 150ms
rememberNode 是否启用节点展开记忆功能(刷新页面后保持展开的节点) boolean false
showTitle 是否启用节点title显示,若设为true,节点对象需有title属性 boolean false
autoHeight tree容器高度是否根据内容自适应 boolean false
dblClickToExpand 是否双击展开节点 boolean true
sort 是否启用节点拖拽排序 boolean false
dataModel {
url:"", //异步加载请求的url，非异步加载不需使用此项
method:"GET", //异步加载方式，非异步加载不需使用此项
dataType:"json", //服务器相应数据类型，非异步加载不需使用此项
params:{}, //需要传递的参数，非异步加载不需使用此项
data:null //数据源，如果是异步加载 此项为空
} 		object dataModel:{}
contextMenu {
items:[{}], //此处为menu的内容数组，格式请参见menu组件
target:"#xxx" //显示menu的dom模型。注：非触发模型
} 		object contextMenu:{}
event {
onClick:function(node){} , //点击节点事件，返回当前点击的节点对象
beforeDrag:function(treeNode){},//treeNode为当前节点数据对象,若返回false停止拖拽
beforeDrop:function(target,source){}, //拖拽时松开鼠标之前执行的事件
onDrop:function(target,source)){},//拖拽之后事件,target(拖拽目标对象),source(拖拽的对象)
beforeExpand:function{node},/节点展开之前事件返回当前节点信息
afterExpand:function{node},//节点展开之后事件返回当前节点信息
onCheck:function{dom},//点击checkbox时执行，返回当前节点对象和其数据对象
afterLoad:function{} //tree加载完成后执行
} 		object event:{}

方法说明

方法名 说明 用例
addNode(newNode(节点对象),parentNodeDom(父节点DOM模型),callback(回调函数)) 增加一个节点方法 var tree = $(dom).awsTree({});tree.addNode(obj1, obj2, callback);
refreshNode(Obj) 节点刷新(Obj={id:父节点id,dataModel:异步刷新dataModel对象,data:子节点数据源} var tree = $(dom).awsTree({});tree.refresh(dataModel);
refresh:function(dataModel) 树的刷新(静态数据设置dataModel.data,异步刷新无需指定data,配置dataModel.url var tree = $(dom).awsTree({});tree.refresh(dataModel);
refreshNodeById(node) 通过节点数据对象刷新对应的节点，node.id为必需 var tree = $(dom).awsTree({});tree.refreshNode({id:id,name:name,cls:cls,title:title,......});
resizeTree() 只有tree的autoHeight属性为false时，才能用于自动填充tree的高度 var tree = $(dom).awsTree({});tree.resizeTree();
selectNode:function(id, expandParent) 选中节点。参数：节点id，是否展开父节点 var tree = $(dom).awsTree({});tree.selectNode(id, expandParent);
cancelSelectNode() 取消选中节点 var tree = $(dom).awsTree({});tree.cancelSelectNode();)
getSelectedNode() 获取选中的节点，返回节点对象 var tree = $(dom).awsTree({});tree.getSelectedNode();
setCheckNodes(ids, isOpen) 设置选中的节点 (参数为id数组和是否自动展开) var nodes = [154,1541,1542];checkTree.setCheckNodes(nodes, true);
getCheckedNodes() checkbox属性为true时,获取checkbox选中的所有节点,返回节点对象数组 var nodes = [154,1541,1542];checkTree.getCheckedNodes();
existsChildren(id)) 通过节点ID验证节点是否已存在子节点 var nodes = [154,1541,1542];checkTree.existsChildren(id);
existsChildren(id)) 通过节点ID验证节点是否已存在子节点 var tree = $(dom).awsTree({});tree.existsChildren(id);
isChecked(obj) checkbox属性为true时判断节点是否被选中，参数为节点对象 var tree = $(dom).awsTree({});tree.isChecked(obj);
getNodeById(id) 通过ID获取节点对象 var tree = $(dom).awsTree({});tree.getNodeById(id);
getNodeDomById(id) 通过ID获取节点的DOM模型 var tree = $(dom).awsTree({});tree.getNodeDomById(id);
getParentNodeById(id) 通过ID获取其父节点对象 var tree = $(dom).awsTree({});tree.getParentNodeById(id);
getRootNode() 获取根节点，返回根节点数组 var tree = $(dom).awsTree({});tree.getRootNode();
getNodeIdsByPid(id) 通过父节点id获取所有子节点的id，返回子节点的id数组 var tree = $(dom).awsTree({});tree.getNodeIdsByPid(id);
getChildrenByPid(id) 通过父节点获取所有的子节点对象 var tree = $(dom).awsTree({});tree.getChildrenByPid(id);
expandNode(obj, keepOpen) 展开一个节点，参数为节点DOM模型、是否保持展开状态 var tree = $(dom).awsTree({});tree.expandNode(obj, keepOpen);
expandNodes(obj(节点模型),expandParent(是否展开父节点),currentStyle(是否包含选中样式),keepOpen(保持展开)) 展开一个节点 var tree = $(dom).awsTree({});tree.expandNode(obj, expandParent,currentStyle,keepOpen);
collapse(obj) 合并一个节点，参数为父节点的DOM模型 var tree = $(dom).awsTree({});tree.collapse(obj);

参数 说明 类型 默认值

nocheck 启用checkbox后，控制节点是否出现checkbox boolean true

dragable 启用拖拽后，若为true不允许被拖拽 boolean true
dropable 启用拖拽后，若为true不允许在此节点上释放 boolean true
iconFont 节点应用字体库图标，iconFont:{code:"图标编码",color:"图标颜色"} object -
icon 节点图标 string -
iconCls 节点图标样式(className与icon互斥) string -
iconOpen 节点展开时的图标 string -/td>
iconClose 节点关闭时的图标 string -

                        JavaScript 代码

                        var dataDrag =[
	{ id:0, name:"蜀国", open:true},
	{ id:1, pid:0, name:"刘备"},
	{ id:2, pid:0, name:"关羽"},
	{ id:3, pid:0, name:"张飞"}
];
var settingDrag = { //拖拽
	sort:true,
	dblClickToExpand:true,
	event:{
		beforeDrag:beforeDrag,
		beforeDrop:beforeDrop,
		onDrop:onDrop
	},
	dataModel:{
		data:dataDrag
	}
};
var dragTree = awsui.tree.init($("#treeDrag"), settingDrag);
function beforeDrag(treeNode){
	console.log("执行beforeDrag事件");
	return true;
}
function beforeDrop(target, source){
	console.log("执行beforeDrop事件");
	return true;
}
function onDrop(target, source, dropSort){
	console.log("拖拽ing");
	if(dropSort = "below"){ //位置调换
		var tempName = source.name; //通常这里需要遍历data数组
		dataDrag[source.id].name = target.name;
		dataDrag[target.id].name = tempName;
	}
	dragTree.refresh({data:dataDrag});
}
