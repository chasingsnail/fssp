气泡框

当目标元素有进一步的描述和相关操作时，可以收纳到卡片中，根据用户的操作行为进行展现。 和 Tooltip 的区别是，用户可以对浮层上的元素进行操作，因此它可以承载更复杂的内容，比如链接或按钮等。

API
  • $(el).popBox(config)

config 参数如下：

参数 说明 类型 默认值
trigger 引发、触发,可选 click、hover、no,若设置为no时，说明已经在其它地方写入了事件，当前不需要再加事件 string click
title popBox 标题 string -
content popBox 内容 string -
width popBox 宽度 string auto
height popBox 高度 string auto
position popBox显示位置,可选 top-right、top、top-left、left、right、bottom-right、bottom、bottom-left string auto
style 组件style,可选 default、inverse string default
hide 点击页面其他地方是否隐藏popbox气泡框 boolean false
delay 鼠标滑动到popbox气泡框区域，多少秒后自动关闭，若设为0不关闭 number 300
基本

最简单的用法，浮层的大小由内容区域决定。

$("#popbox-dialog").on("click", function(){
	$("#dialog-nomask").dialog({
		model:true,
		width:800,
		height:200,
		buttons:[
			{text:'确定',handler:function(e){
					e.stopPropagation();

						trigger:'no',
						title:'popBox 标题',
						content:'

',
						hide:true
					});
				}}
		]
	});
});
popBox中嵌套js创建的DOM元素

popBox中嵌套js创建的DOM元素。

HTML 代码

<button type="button" class="awsui-btn awsui-btn-blue" id="popbox-dom">点击</button>

JavaScript 代码

$("#popbox-dom").popBox({
	title:'Dom元素加载',
	width: 420,
	height: 200,
	delay:0,
	content:function () {
		var index = $(this).attr("index") || 0;
		return renderMemberList(index);
	}
});
function renderMemberList(index){
	//var memberList = appTeamList[0][index]["members"];
	var memberList = [{deptName: "信息中心",name: "管理员",uid: "admin"},{deptName: "消息中心",name: "管理员",uid: "admin"},{deptName: "消息中心",name: "管理员",uid: "admin"},{deptName: "消息中心",name: "管理员",uid: "admin"},{deptName: "消息中心",name: "管理员",uid: "admin"}];
	var listHtml="";
	if(memberList.length>0){
		listHtml+='<table class="awsui-table awsui-table-bordered" width="420" border="0" cellspacing="0" cellpadding="0">';
		listHtml+='<tr>';
		listHtml+='<td width="20%">姓名</td>';
		listHtml+='<td width="20%">账号</td>';
		listHtml+='<td width="60%">部门</td>';
		listHtml+='</tr>';
		for (var i = 0;i < memberList.length;i++){
			var memberInfo =  memberList[i];
			var name = memberInfo["name"];
			var uid = memberInfo["uid"];
			var deptName = memberInfo["deptName"];
			listHtml+='<tr>';
			listHtml+='<td><div class="ellipsistext td1text" awsui-qtip="'+name+'">'+name+'</div></td>';
			listHtml+='<td><div class="ellipsistext td2text" awsui-qtip="'+uid+'">'+uid+'</div></td>';
			listHtml+='<td><div class="ellipsistext td3text" awsui-qtip="'+deptName+'">'+deptName+'</div></td>';
			listHtml+='</tr>';
		}
		listHtml+='</table>';
	}
	return listHtml;
}
