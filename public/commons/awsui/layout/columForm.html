表单

具有数据收集、校验和提交功能的表单，包含复选框、单选框、输入框、下拉选择框等元素，我们为 form 提供了以下三种布局方式：

  • 水平布局：标签和表单控件水平排列。
  • 垂直布局：标签和表单控件上下垂直排列。
  • 行内两列布局：表单项水平行内排列。

$(document).ready(function() {
	awsui.parse(); //日历插件
	var opt = {
		width : "100%",
		data : [
			{id:"lb",text:"刘备"},
			{id:"gy",text:"关羽"},
			{id:"zf",text:"张飞"}
		]
	};
	$("#demo1").select2(opt);
	$("#buttonedit1").on("click",function(){
		alert("这是一个搜索框");
	});
	$("#switch").switchButton();
	$("#submit1").on("click",function(){
		$.simpleAlert("点击了提交","ok");
	});
	$("#cancel1").on("click",function(){
		$.simpleAlert("点击了取消");
	});
	$(".required").find(".awsui-input").focus(function () {
		$.each($(".required").find(".awsui-input"),function(i,el){
			$(el).css("background-color","#fff");
		});
		$(this).css("background-color","#FFFFCC");
	});
	$(".required").find(".awsui-input").blur(function () {
		var inputVal = $(this).val();
		if (inputVal == "") {
			$(this).tooltip({
				positon : "top",
				text : "不允许为空，请重新输入",
				bgcolor : "#f9f9f9",
				color : "red",
				borderColor : "red",
				close:false
			});
			return false;
		}
	});
});
